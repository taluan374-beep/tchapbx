main.py Python 3.11 
# Initiate an outbound call with HaPBX
from hapbx import Client

client = Client("your_api_key")

# Make a call with real-time transcription
call = client.calls.create(
    to="+1-555-123-4567",
    from_="+1-555-987-6543",
    webhook="https://app.example.com/events",
    record=True,
    transcribe=True
)

print(f"Call initiated: {call.sid}")
# Output: Call initiated: CA123abc456def789
